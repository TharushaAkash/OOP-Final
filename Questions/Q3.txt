package Final4.Q3;

import java.util.Scanner;

public class InvalidProductCode extends Exception{
    public InvalidProductCode() {
        super("Product code must starts with 'P' and max length is 6 and ends with 'E'");
    }
}

class InvalidPriceRange extends Exception{
    public InvalidPriceRange() {
        super("Product price must be greater than $0.49");
    }
}

class ProductRegistrationApp {
    public static void main(String[] args) {
        Scanner sc= new Scanner(System.in);


        System.out.print("Enter Product Code: ");
        String pCode = sc.nextLine();

        try {
            if((pCode.charAt(0) != 'P' || pCode.charAt(5) != 'E') || pCode.length()  > 6 ) {
                throw new InvalidProductCode();
            }
            System.out.print("Enter product Price: ");
            double price = sc.nextDouble();

            try {
                if(price <= 0.5){
                    throw new InvalidPriceRange();
                }
                System.out.println("Product registered successfully!");
            }catch (InvalidPriceRange p){
                System.out.println(p.getMessage());
            }

        }catch (InvalidProductCode p){
            System.out.println(p.getMessage());

        }




    }
}

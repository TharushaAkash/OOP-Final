import java.util.Scanner;

public class Book {
    private String bookId;
    private String bookTitle;
    private String author;
    private int copiesAvailable;

    public Book(String bookId, String bookTitle, String author, int copiesAvailable) {
        this.bookId = bookId;
        this.bookTitle = bookTitle;
        this.author = author;
        this.copiesAvailable = copiesAvailable;
    }

    void borrowBook(int copies){
        if(copies > copiesAvailable){
            System.out.println("Insufficient copies available. Please request fewer copies or try another book.");
        }
        else{
            copiesAvailable -= copies;
            System.out.println("Book borrowed successfully!");
        }
    }

    void returnBook(int returnd){
        copiesAvailable += returnd;
        System.out.println("Book returned successfully!");
    }

    void displayBookDetails(){
        System.out.println("Book ID: " + bookId);
        System.out.println("Title: " + bookTitle);
        System.out.println("Author: " + author);
        System.out.println("Copies Available: " + copiesAvailable);
        System.out.println("-------------------------------------");
    }
}

class LibraryApp{
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        Book[]books = new Book[3];

        for(int i = 0; i < books.length; i++){
            System.out.println("Enter details for Book " + (i+1) + ": ");
            System.out.print("Book ID: ");
            String id = sc.nextLine();

            System.out.print("Title: ");
            String title = sc.nextLine();

            System.out.print("Author: ");
            String author = sc.nextLine();

            System.out.print("Copies Available: ");
            int copies = sc.nextInt();
            sc.nextLine();//Remove bufferspace
            System.out.println(); //Line Space

            Book book = new Book(id, title, author, copies);
            books[i] = book;
        }


        System.out.print("Enter number of copies to borrow for Book 1: ");
        int copiesB = sc.nextInt();
        books[0].borrowBook(copiesB);
        System.out.println(); //Line Space

        System.out.print("Enter number of copies to borrow for Book 2: ");
        int copiesB0= sc.nextInt();
        books[1].borrowBook(copiesB0);
        System.out.println();

        System.out.print("Enter number of copies to borrow for Book 3: ");
        int copiesB1 = sc.nextInt();
        books[2].borrowBook(copiesB1);
        System.out.println();

        System.out.print("Enter number of copies to return for Book 1: ");
        int copiesR = sc.nextInt();
        books[0].returnBook(copiesR);
        System.out.println();
        //line space
        System.out.println("Final Book Details:");
        for(Book b : books){
            b.displayBookDetails();
        }
    }
}
